44/// Helper function to send GetFileRefs instruction and wait for reply.
45async fn send_get_file_refs(
46    mailbox: &MailboxRef,
47    server_addr: &str,
48    peer: &Peer,
49    instruction: GetFileRefs,
50    timeout: Duration,
51) -> Result<GetFileRefsReply> {
52    let instruction = instruction::Instruction::GetFileRefs(instruction);
53    let msg = MailboxMessage::json_message(
54        &format!("Get file references: {}", instruction),
55        &format!("Metasrv@{}", server_addr),
56        &format!("Datanode-{}@{}", peer.id, peer.addr),
57        common_time::util::current_time_millis(),
58        &instruction,
59    )
60    .with_context(|_| SerializeToJsonSnafu {
61        input: instruction.to_string(),
62    })?;
63
64    let mailbox_rx = mailbox
65        .send(&Channel::Datanode(peer.id), msg, timeout)
66        .await?;
67
68    let reply = match mailbox_rx.await {
69        Ok(reply_msg) => HeartbeatMailbox::json_reply(&reply_msg)?,
70        Err(e) => {
71            error!(
72                "Failed to receive reply from datanode {} for GetFileRefs: {}",
73                peer, e
74            );
75            return Err(e);
76        }
77    };
78
79    let InstructionReply::GetFileRefs(reply) = reply else {
80        return error::UnexpectedInstructionReplySnafu {
81            mailbox_message: format!("{:?}", reply),
82            reason: "Unexpected reply of the GetFileRefs instruction",
83        }
84        .fail();
85    };
86
87    Ok(reply)
88}

90/// Helper function to send GcRegions instruction and wait for reply.
91async fn send_gc_regions(
92    mailbox: &MailboxRef,
93    peer: &Peer,
94    gc_regions: GcRegions,
95    server_addr: &str,
96    timeout: Duration,
97    description: &str,
98) -> Result<GcReport> {
99    let instruction = instruction::Instruction::GcRegions(gc_regions.clone());
100    let msg = MailboxMessage::json_message(
101        &format!("{}: {}", description, instruction),
102        &format!("Metasrv@{}", server_addr),
103        &format!("Datanode-{}@{}", peer.id, peer.addr),
104        common_time::util::current_time_millis(),
105        &instruction,
106    )
107    .with_context(|_| SerializeToJsonSnafu {
108        input: instruction.to_string(),
109    })?;
110
111    let mailbox_rx = mailbox
112        .send(&Channel::Datanode(peer.id), msg, timeout)
113        .await?;
114
115    let reply = match mailbox_rx.await {
116        Ok(reply_msg) => HeartbeatMailbox::json_reply(&reply_msg)?,
117        Err(e) => {
118            error!(
119                "Failed to receive reply from datanode {} for {}: {}",
120                peer, description, e
121            );
122            return Err(e);
123        }
124    };
125
126    let InstructionReply::GcRegions(reply) = reply else {
127        return error::UnexpectedInstructionReplySnafu {
128            mailbox_message: format!("{:?}", reply),
129            reason: "Unexpected reply of the GcRegions instruction",
130        }
131        .fail();
132    };
133
134    let res = reply.result;
135    match res {
136        Ok(report) => Ok(report),
137        Err(e) => {
138            error!(
139                "Datanode {} reported error during GC for regions {:?}: {}",
140                peer, gc_regions, e
141            );
142            error::UnexpectedSnafu {
143                violated: format!(
144                    "Datanode {} reported error during GC for regions {:?}: {}",
145                    peer, gc_regions, e
146                ),
147            }
148            .fail()
149        }
150    }
151}

153/// Procedure to perform get file refs then batch GC for multiple regions,
154/// it holds locks for all regions during the whole procedure.
155pub struct BatchGcProcedure {
156    mailbox: MailboxRef,
157    table_metadata_manager: TableMetadataManagerRef,
158    data: BatchGcData,
159}
160
161#[derive(Serialize, Deserialize)]
162pub struct BatchGcData {
163    state: State,
164    /// Meta server address
165    server_addr: String,
166    /// The regions to be GC-ed
167    regions: Vec<RegionId>,
168    full_file_listing: bool,
169    region_routes: Region2Peers,
170    /// Related regions (e.g., for shared files after repartition).
171    /// The source regions (where those files originally came from) are used as the key, and the destination regions (where files are currently stored) are used as the value.
172    related_regions: HashMap<RegionId, HashSet<RegionId>>,
173    /// Acquired file references (Populated in Acquiring state)
174    file_refs: FileRefsManifest,
175    /// mailbox timeout duration
176    timeout: Duration,
177    gc_report: Option<GcReport>,
178}
179
180#[derive(Debug, Clone, Serialize, Deserialize, PartialEq)]
181pub enum State {
182    /// Initial state
183    Start,
184    /// Fetching file references from datanodes
185    Acquiring,
186    /// Sending GC instruction to the target datanode
187    Gcing,
188    /// Updating region repartition info in kvbackend after GC based on the GC result
189    UpdateRepartition,
190}

232    async fn get_table_route(

243    /// Return related regions for the given regions.
244    /// The returned map uses the source regions (where those files originally came from) as the key,
245    /// and the destination regions (where files are currently stored) as the value.
246    /// If a region is not found in the repartition manager, the returned map still have this region as key,
247    /// just empty value
248    async fn find_related_regions(
249        &self,
250        regions: &[RegionId],
251    ) -> Result<HashMap<RegionId, HashSet<RegionId>>> {
252        let repart_mgr = self.table_metadata_manager.table_repart_manager();
253        let mut related_regions: HashMap<RegionId, HashSet<RegionId>> = HashMap::new();
254        for src_region in regions {
255            // TODO(discord9): batch get
256            if let Some(dst_regions) = repart_mgr
257                .get_dst_regions(*src_region)
258                .await
259                .context(KvBackendSnafu)?
260            {
261                related_regions.insert(*src_region, dst_regions.into_iter().collect());
262            } else {
263                related_regions.insert(*src_region, Default::default());
264            }
265        }
266        Ok(related_regions)
267    }

269    /// Clean up region repartition info in kvbackend after GC

271    async fn cleanup_region_repartition(&self) -> Result<()> {
272        for (src_region, dst_regions) in self.data.file_refs.cross_region_refs.iter() {

278                .context(KvBackendSnafu)?;
279        }
280        Ok(())

283    /// Discover region routes for the given regions.
284    async fn discover_route_for_regions(
285        &self,
286        regions: &[RegionId],
287    ) -> Result<(Region2Peers, Peer2Regions)> {
288        let mut region_to_peer = HashMap::new();
289        let mut peer_to_regions = HashMap::new();
290
291        // Group regions by table ID for batch processing
292        let mut table_to_regions: HashMap<TableId, Vec<RegionId>> = HashMap::new();
293        for region_id in regions {
294            let table_id = region_id.table_id();
295            table_to_regions
296                .entry(table_id)
297                .or_default()
298                .push(*region_id);
299        }
300
301        // Process each table's regions together for efficiency
302        for (table_id, table_regions) in table_to_regions {
303            match self.get_table_route(table_id).await {
304                Ok((_phy_table_id, table_route)) => {
305                    table_route_to_region(
306                        &table_route,
307                        &table_regions,
308                        &mut region_to_peer,
309                        &mut peer_to_regions,
310                    );
311                }
312                Err(e) => {
313                    // Continue with other tables instead of failing completely
314                    // TODO(discord9): consider failing here instead
315                    warn!(
316                        "Failed to get table route for table {}: {}, skipping its regions",
317                        table_id, e
318                    );
319                    continue;
320                }
321            }
322        }
323
324        Ok((region_to_peer, peer_to_regions))
325    }

327    /// Set region routes and related regions for GC procedure
328    async fn set_routes_and_related_regions(&mut self) -> Result<()> {
329        let related_regions = self.find_related_regions(&self.data.regions).await?;
330
331        self.data.related_regions = related_regions.clone();
332
333        // Discover routes for all regions involved in GC, including both the
334        // primary GC regions and their related regions.
335        let mut regions_set: HashSet<RegionId> = self.data.regions.iter().cloned().collect();
336
337        regions_set.extend(related_regions.keys().cloned());
338        regions_set.extend(related_regions.values().flat_map(|v| v.iter()).cloned());
339
340        let regions_to_discover = regions_set.into_iter().collect_vec();
341
342        let (region_to_peer, _) = self
343            .discover_route_for_regions(&regions_to_discover)
344            .await?;
345
346        self.data.region_routes = region_to_peer;
347
348        Ok(())
349    }

351    /// Get file references from all datanodes that host the regions
352    async fn get_file_references(&mut self) -> Result<FileRefsManifest> {
353        self.set_routes_and_related_regions().await?;
354
355        let query_regions = &self.data.regions;
356        let related_regions = &self.data.related_regions;
357        let region_routes = &self.data.region_routes;
358        let timeout = self.data.timeout;
359
360        // Group regions by datanode to minimize RPC calls
361        let mut datanode2query_regions: HashMap<Peer, Vec<RegionId>> = HashMap::new();
362
363        for region_id in query_regions {
364            if let Some((leader, followers)) = region_routes.get(region_id) {
365                datanode2query_regions
366                    .entry(leader.clone())
367                    .or_default()
368                    .push(*region_id);
369                // also need to send for follower regions for file refs in case query is running on follower
370                for follower in followers {
371                    datanode2query_regions
372                        .entry(follower.clone())
373                        .or_default()
374                        .push(*region_id);
375                }
376            } else {
377                return error::UnexpectedSnafu {
378                    violated: format!(
379                        "region_routes: {region_routes:?} does not contain region_id: {region_id}",
380                    ),
381                }
382                .fail();
383            }
384        }
385
386        let mut datanode2related_regions: HashMap<Peer, HashMap<RegionId, HashSet<RegionId>>> =
387            HashMap::new();
388        for (src_region, dst_regions) in related_regions {
389            for dst_region in dst_regions {
390                if let Some((leader, _followers)) = region_routes.get(dst_region) {
391                    datanode2related_regions
392                        .entry(leader.clone())
393                        .or_default()
394                        .entry(*src_region)
395                        .or_default()
396                        .insert(*dst_region);
397                } // since read from manifest, no need to send to followers
398            }
399        }
400
401        // Send GetFileRefs instructions to each datanode
402        let mut all_file_refs: HashMap<RegionId, HashSet<_>> = HashMap::new();
403        let mut all_manifest_versions = HashMap::new();
404        let mut all_cross_region_refs = HashMap::new();
405
406        for (peer, regions) in datanode2query_regions {
407            let related_regions_for_peer =
408                datanode2related_regions.remove(&peer).unwrap_or_default();
409
410            let instruction = GetFileRefs {
411                query_regions: regions.clone(),
412                related_regions: related_regions_for_peer,
413            };
414
415            let reply = send_get_file_refs(
416                &self.mailbox,
417                &self.data.server_addr,
418                &peer,
419                instruction,
420                timeout,
421            )
422            .await?;
423
424            if !reply.success {
425                return error::UnexpectedSnafu {
426                    violated: format!(
427                        "Failed to get file references from datanode {}: {:?}",
428                        peer, reply.error
429                    ),
430                }
431                .fail();
432            }
433
434            // Merge the file references from this datanode
435            for (region_id, file_refs) in reply.file_refs_manifest.file_refs {
436                all_file_refs
437                    .entry(region_id)
438                    .or_default()
439                    .extend(file_refs);
440            }
441
442            // region manifest version should be the smallest one among all peers, so outdated region can be detected
443            for (region_id, version) in reply.file_refs_manifest.manifest_version {
444                let entry = all_manifest_versions.entry(region_id).or_insert(version);
445                *entry = (*entry).min(version);
446            }
447
448            for (region_id, related_region_ids) in reply.file_refs_manifest.cross_region_refs {
449                let entry = all_cross_region_refs
450                    .entry(region_id)
451                    .or_insert_with(HashSet::new);
452                entry.extend(related_region_ids);
453            }
454        }
455
456        Ok(FileRefsManifest {
457            file_refs: all_file_refs,
458            manifest_version: all_manifest_versions,
459            cross_region_refs: all_cross_region_refs,
460        })
461    }

463    /// Send GC instruction to all datanodes that host the regions,
464    /// returns regions that need retry.
465    async fn send_gc_instructions(&self) -> Result<GcReport> {
466        let regions = &self.data.regions;
467        let region_routes = &self.data.region_routes;
468        let file_refs = &self.data.file_refs;
469        let timeout = self.data.timeout;
470
471        // Group regions by datanode
472        let mut datanode2regions: HashMap<Peer, Vec<RegionId>> = HashMap::new();
473        let mut all_report = GcReport::default();
474
475        for region_id in regions {
476            if let Some((leader, _followers)) = region_routes.get(region_id) {
477                datanode2regions
478                    .entry(leader.clone())
479                    .or_default()
480                    .push(*region_id);
481            } else {
482                return error::UnexpectedSnafu {
483                    violated: format!(
484                        "region_routes: {region_routes:?} does not contain region_id: {region_id}",
485                    ),
486                }
487                .fail();
488            }
489        }
490
491        let mut all_need_retry = HashSet::new();
492        // Send GC instructions to each datanode
493        for (peer, regions_for_peer) in datanode2regions {
494            let gc_regions = GcRegions {
495                regions: regions_for_peer.clone(),
496                // file_refs_manifest can be large; cloning for each datanode is acceptable here since this is an admin-only operation.
497                file_refs_manifest: file_refs.clone(),
498                full_file_listing: self.data.full_file_listing,
499            };
500
501            let report = send_gc_regions(
502                &self.mailbox,
503                &peer,
504                gc_regions,
505                self.data.server_addr.as_str(),
506                timeout,
507                "Batch GC",
508            )
509            .await?;
510
511            let success = report.deleted_files.keys().collect_vec();
512            let need_retry = report.need_retry_regions.iter().cloned().collect_vec();
513
514            if need_retry.is_empty() {
515                info!(
516                    "GC report from datanode {}: successfully deleted files for regions {:?}",
517                    peer, success
518                );
519            } else {
520                warn!(
521                    "GC report from datanode {}: successfully deleted files for regions {:?}, need retry for regions {:?}",
522                    peer, success, need_retry
523                );
524            }
525            all_need_retry.extend(report.need_retry_regions.clone());
526            all_report.merge(report);
527        }
528
529        if !all_need_retry.is_empty() {
530            warn!("Regions need retry after batch GC: {:?}", all_need_retry);
531        }
532
533        Ok(all_report)
534    }
535}

537#[async_trait::async_trait]
538impl Procedure for BatchGcProcedure {
539    fn type_name(&self) -> &str {
540        Self::TYPE_NAME
541    }
542
543    async fn execute(&mut self, _ctx: &ProcedureContext) -> ProcedureResult<Status> {
544        match self.data.state {
545            State::Start => {
546                // Transition to Acquiring state
547                self.data.state = State::Acquiring;
548                Ok(Status::executing(false))
549            }
550            State::Acquiring => {
551                // Get file references from all datanodes
552                match self.get_file_references().await {
553                    Ok(file_refs) => {
554                        self.data.file_refs = file_refs;
555                        self.data.state = State::Gcing;
556                        Ok(Status::executing(false))
557                    }
558                    Err(e) => {
559                        error!("Failed to get file references: {}", e);
560                        Err(ProcedureError::external(e))
561                    }
562                }
563            }
564            State::Gcing => {
565                // Send GC instructions to all datanodes
566                // TODO(discord9): handle need-retry regions
567                match self.send_gc_instructions().await {
568                    Ok(report) => {
569                        self.data.state = State::UpdateRepartition;
570                        self.data.gc_report = Some(report);
571                        Ok(Status::executing(false))
572                    }
573                    Err(e) => {
574                        error!("Failed to send GC instructions: {}", e);
575                        Err(ProcedureError::external(e))
576                    }
577                }
578            }
579            State::UpdateRepartition => match self.cleanup_region_repartition().await {
580                Ok(()) => {
581                    info!(
582                        "Cleanup region repartition info completed successfully for regions {:?}",
583                        self.data.regions
584                    );
585                    info!(
586                        "Batch GC completed successfully for regions {:?}",
587                        self.data.regions
588                    );
589                    let Some(report) = self.data.gc_report.take() else {
590                        return common_procedure::error::UnexpectedSnafu {
591                            err_msg: "GC report should be present after GC completion".to_string(),
592                        }
593                        .fail();
594                    };
595                    Ok(Status::done_with_output(report))
596                }
597                Err(e) => {
598                    error!("Failed to cleanup region repartition info: {}", e);
599                    Err(ProcedureError::external(e))
600                }
601            },
602        }
603    }
604
605    fn dump(&self) -> ProcedureResult<String> {
606        serde_json::to_string(&self.data).context(ToJsonSnafu)
607    }

609    /// Read lock all regions involved in this GC procedure.
610    /// So i.e. region migration won't happen during GC and cause race conditions.
611    fn lock_key(&self) -> LockKey {
612        let lock_key: Vec<_> = self
613            .data
614            .regions
615            .iter()
616            .sorted() // sort to have a deterministic lock order
617            .map(|id| RegionLock::Read(*id).into())
618            .collect();
619
620        LockKey::new(lock_key)
621    }
