34pub(crate) fn derive_to_row_impl(input: DeriveInput) -> Result<TokenStream2> {
35    let Data::Struct(DataStruct {
36        fields: Fields::Named(FieldsNamed { named, .. }),
37        ..
38    }) = &input.data
39    else {
40        return Err(syn::Error::new(
41            input.span(),
42            "ToRow can only be derived for structs",
43        ));
44    };
45
46    let ident = input.ident;
47    let generics = input.generics;
48    let (impl_generics, ty_generics, where_clause) = generics.split_for_impl();
49
50    let fields = named.iter().map(|field| {
51        (
52            field.ident.as_ref().expect("field must have an ident"),
53            &field.ty,
54        )
55    });
56
57    // Ident of fields.
58    let idents = fields.clone().map(|(ident, _)| ident).collect::<Vec<_>>();
59    // Infer column data types for each field.
60    let infer_column_data_types = fields
61        .clone()
62        .map(|(_, ty)| field_type(ty))
63        .flat_map(|ty| {
64            ty.extract_ident()
65                .map(convert_primitive_type_to_column_data_type)
66        })
67        .collect::<Vec<_>>();
68
69    // Find attributes for each field.
70    let column_attributes = named
71        .iter()
72        .map(|field| {
73            let attrs = &field.attrs;
74            let attr = find_column_attribute(attrs);
75            match attr {
76                Some(attr) => parse_attribute(attr),
77                None => Ok(ColumnAttribute::default()),
78            }
79        })
80        .collect::<Result<Vec<ColumnAttribute>>>()?;
81    let field_types = fields
82        .clone()
83        .map(|(_, ty)| field_type(ty))
84        .collect::<Vec<_>>();
85
86    // Implement `to_row` method.
87    let impl_to_row_method = impl_to_row_method_combined(
88        &idents,
89        &field_types,
90        &infer_column_data_types,
91        &column_attributes,
92    )?;
93
94    // Implement `schema` method.
95    let impl_schema_method =
96        impl_schema_method(&idents, &infer_column_data_types, &column_attributes)?;
97
98    Ok(quote! {
99        impl #impl_generics #ident #ty_generics #where_clause {
100            #impl_to_row_method
101
102            #impl_schema_method
103        }
104    })
105}

107fn impl_schema_method(
108    idents: &[&Ident],
109    infer_column_data_types: &[Option<ColumnDataType>],
110    column_attributes: &[ColumnAttribute],
111) -> Result<TokenStream2> {
112    let schemas: Vec<TokenStream2> = idents
113        .iter()
114        .zip(infer_column_data_types.iter())
115        .zip(column_attributes.iter())
116        .map(|((ident, column_data_type), column_attribute)| {
117            let Some(column_data_type) = get_column_data_type(column_data_type, column_attribute)
118            else {
119                return Err(syn::Error::new(
120                    ident.span(),
121                    format!(
122                        "expected to set data type explicitly via [({META_KEY_COL}({META_KEY_DATATYPE} = \"...\"))]"
123                    ),
124                ));
125            };
126
127            // Uses user explicit name or field name as column name.
128            let name = column_attribute
129                .name
130                .clone()
131                .unwrap_or_else(|| ident.to_string());
132            let name = syn::LitStr::new(&name, ident.span());
133            let column_data_type =
134                syn::LitInt::new(&(column_data_type as i32).to_string(), ident.span());
135            let semantic_type_val = match column_attribute.semantic_type {
136                SemanticType::Field => greptime_proto::v1::SemanticType::Field,
137                SemanticType::Tag => greptime_proto::v1::SemanticType::Tag,
138                SemanticType::Timestamp => greptime_proto::v1::SemanticType::Timestamp,
139            } as i32;
140            let semantic_type = syn::LitInt::new(&semantic_type_val.to_string(), ident.span());
141
142            Ok(quote! {
143                greptime_proto::v1::ColumnSchema {
144                    column_name: #name.to_string(),
145                    datatype: #column_data_type,
146                    datatype_extension: None,
147                    options: None,
148                    semantic_type: #semantic_type,
149                }
150            })
151        })
152        .collect::<Result<_>>()?;
153
154    Ok(quote! {
155        pub fn schema(&self) -> Vec<greptime_proto::v1::ColumnSchema> {
156            vec![ #(#schemas),* ]
157        }
158    })
159}

161fn impl_to_row_method_combined(
162    idents: &[&Ident],
163    field_types: &[FieldType<'_>],
164    infer_column_data_types: &[Option<ColumnDataType>],
165    column_attributes: &[ColumnAttribute],
166) -> Result<TokenStream2> {
167    let value_exprs = idents
168        .iter()
169        .zip(field_types.iter())
170        .zip(infer_column_data_types.iter())
171        .zip(column_attributes.iter())
172        .map(|(((ident, field_type), column_data_type), column_attribute)| {
173            let Some(column_data_type) = get_column_data_type(column_data_type, column_attribute)
174            else {
175                return Err(syn::Error::new(
176                    ident.span(),
177                    format!(
178                        "expected to set data type explicitly via [({META_KEY_COL}({META_KEY_DATATYPE} = \"...\"))]"
179                    ),
180                ));
181            };
182            let value_data = convert_column_data_type_to_value_data_ident(&column_data_type);
183            let expr = if field_type.is_optional() {
184                quote! {
185                    match &self.#ident {
186                        Some(v) => greptime_proto::v1::Value {
187                            value_data: Some(greptime_proto::v1::value::ValueData::#value_data(v.clone().into())),
188                        },
189                        None => greptime_proto::v1::Value { value_data: None },
190                    }
191                }
192            } else {
193                quote! {
194                    greptime_proto::v1::Value {
195                        value_data: Some(greptime_proto::v1::value::ValueData::#value_data(self.#ident.clone().into())),
196                    }
197                }
198            };
199            Ok(expr)
200        })
201        .collect::<Result<Vec<_>>>()?;
202
203    Ok(quote! {
204        pub fn to_row(&self) -> greptime_proto::v1::Row {
205            greptime_proto::v1::Row {
206                values: vec![ #( #value_exprs ),* ]
207            }
208        }
209    })
210}

212fn get_column_data_type(
213    infer_column_data_type: &Option<ColumnDataType>,
214    attribute: &ColumnAttribute,
215) -> Option<ColumnDataType> {
216    attribute.column_data_type.or(*infer_column_data_type)
217}

219fn convert_column_data_type_to_value_data_ident(column_data_type: &ColumnDataType) -> Ident {
220    match column_data_type {
221        ColumnDataType::Boolean => format_ident!("BoolValue"),
222        ColumnDataType::Int8 => format_ident!("I8Value"),
223        ColumnDataType::Int16 => format_ident!("I16Value"),
224        ColumnDataType::Int32 => format_ident!("I32Value"),
225        ColumnDataType::Int64 => format_ident!("I64Value"),
226        ColumnDataType::Uint8 => format_ident!("U8Value"),
227        ColumnDataType::Uint16 => format_ident!("U16Value"),
228        ColumnDataType::Uint32 => format_ident!("U32Value"),
229        ColumnDataType::Uint64 => format_ident!("U64Value"),
230        ColumnDataType::Float32 => format_ident!("F32Value"),
231        ColumnDataType::Float64 => format_ident!("F64Value"),
232        ColumnDataType::Binary => format_ident!("BinaryValue"),
233        ColumnDataType::String => format_ident!("StringValue"),
234        ColumnDataType::Date => format_ident!("DateValue"),
235        ColumnDataType::Datetime => format_ident!("DatetimeValue"),
236        ColumnDataType::TimestampSecond => format_ident!("TimestampSecondValue"),
237        ColumnDataType::TimestampMillisecond => {
238            format_ident!("TimestampMillisecondValue")
239        }
240        ColumnDataType::TimestampMicrosecond => {
241            format_ident!("TimestampMicrosecondValue")
242        }
243        ColumnDataType::TimestampNanosecond => format_ident!("TimestampNanosecondValue"),
244        ColumnDataType::TimeSecond => format_ident!("TimeSecondValue"),
245        ColumnDataType::TimeMillisecond => format_ident!("TimeMillisecondValue"),
246        ColumnDataType::TimeMicrosecond => format_ident!("TimeMicrosecondValue"),
247        ColumnDataType::TimeNanosecond => format_ident!("TimeNanosecondValue"),
248        ColumnDataType::IntervalYearMonth => format_ident!("IntervalYearMonthValue"),
249        ColumnDataType::IntervalDayTime => format_ident!("IntervalDayTimeValue"),
250        ColumnDataType::IntervalMonthDayNano => {
251            format_ident!("IntervalMonthDayNanoValue")
252        }
253        ColumnDataType::Decimal128 => format_ident!("Decimal128Value"),
254        ColumnDataType::Json => format_ident!("JsonValue"),
255        ColumnDataType::Vector => format_ident!("VectorValue"),
256    }
257}

259fn field_type(ty: &Type) -> FieldType<'_> {
260    if let Type::Reference(TypeReference { elem, .. }) = ty {
261        return field_type(elem);
262    }
263
264    if let Type::Path(TypePath {
265        qself: _,
266        path: Path {
267            leading_colon,
268            segments,
269        },
270    }) = ty
271    {
272        if leading_colon.is_none() && segments.len() == 1 {
273            if let Some(PathSegment {
274                ident,
275                arguments:
276                    PathArguments::AngleBracketed(AngleBracketedGenericArguments { args, .. }),
277            }) = segments.first()
278            {
279                if let (1, Some(GenericArgument::Type(t))) = (args.len(), args.first()) {
280                    if ident == "Option" {
281                        return FieldType::Optional(t);
282                    }
283                }
284            }
285        }
286    }
287
288    FieldType::Required(ty)
289}

291fn extract_ident_from_type(ty: &Type) -> Option<&Ident> {
292    match ty {
293        Type::Path(TypePath { qself: None, path }) => path.get_ident(),
294        Type::Reference(type_ref) => extract_ident_from_type(&type_ref.elem),
295        Type::Group(type_group) => extract_ident_from_type(&type_group.elem),
296        _ => None,
297    }
298}

347static SEMANTIC_TYPES: Lazy<HashMap<&'static str, SemanticType>> = Lazy::new(|| {
348    HashMap::from([
349        ("field", SemanticType::Field),
350        ("tag", SemanticType::Tag),
351        ("timestamp", SemanticType::Timestamp),

355fn convert_primitive_type_to_column_data_type(ident: &Ident) -> Option<ColumnDataType> {
356    PRIMITIVE_TYPE_TO_COLUMN_DATA_TYPE
357        .get(ident.to_string().as_str())
358        .cloned()
359}

361fn semantic_type_from_ident(ident: &str) -> Option<SemanticType> {
362    // Ignores the case of the identifier.
363    let lowercase = ident.to_lowercase();
364    let lowercase_str = lowercase.as_str();
365    SEMANTIC_TYPES.get(lowercase_str).cloned()
366}
367
368fn convert_field_type_to_column_data_type(ident: &str) -> Option<ColumnDataType> {
369    // Ignores the case of the identifier.
370    let lowercase = ident.to_lowercase();
371    let lowercase_str = lowercase.as_str();
372    DATATYPE_TO_COLUMN_DATA_TYPE.get(lowercase_str).cloned()
373}

375#[derive(Default, Clone, Copy)]
376enum SemanticType {
377    #[default]
378    Field,
379    Tag,
380    Timestamp,
381}
382
383enum FieldType<'a> {
384    Required(&'a Type),
385    Optional(&'a Type),
386}
387
388impl FieldType<'_> {
389    fn is_optional(&self) -> bool {
390        matches!(self, FieldType::Optional(_))
391    }
392
393    fn extract_ident(&self) -> Option<&Ident> {
394        match self {
395            FieldType::Required(ty) => extract_ident_from_type(ty),
396            FieldType::Optional(ty) => extract_ident_from_type(ty),
397        }
398    }
399}

401#[derive(Default)]
402struct ColumnAttribute {
403    name: Option<String>,
404    column_data_type: Option<ColumnDataType>,
405    semantic_type: SemanticType,
406}
407
408fn find_column_attribute(attrs: &[Attribute]) -> Option<&Attribute> {
409    attrs
410        .iter()
411        .find(|attr| matches!(&attr.meta, Meta::List(list) if list.path.is_ident(META_KEY_COL)))
412}
413
414fn parse_attribute(attr: &Attribute) -> Result<ColumnAttribute> {
415    match &attr.meta {
416        Meta::List(list) if list.path.is_ident(META_KEY_COL) => {
417            let mut attribute = ColumnAttribute::default();
418            list.parse_nested_meta(|meta| {
419                parse_column_attribute(&meta, &mut attribute)
420            })?;
421            Ok(attribute)
422        }
423        _ => Err(syn::Error::new(
424            attr.span(),
425            format!(
426                "expected `{META_KEY_COL}({META_KEY_NAME} = \"...\", {META_KEY_DATATYPE} = \"...\", {META_KEY_SEMANTIC} = \"...\")`"
427            ),
428        )),
429    }
430}
431
432fn parse_column_attribute(
433    meta: &ParseNestedMeta<'_>,
434    attribute: &mut ColumnAttribute,
435) -> Result<()> {
436    let Some(ident) = meta.path.get_ident() else {
437        return Err(meta.error(format!("expected `{META_KEY_COL}({META_KEY_NAME} = \"...\", {META_KEY_DATATYPE} = \"...\", {META_KEY_SEMANTIC} = \"...\")`")));
438    };
439
440    match ident.to_string().as_str() {
441        META_KEY_NAME => {
442            let value = meta.value()?;
443            let s: LitStr = value.parse()?;
444            attribute.name = Some(s.value());
445        }
446        META_KEY_DATATYPE => {
447            let value = meta.value()?;
448            let s: LitStr = value.parse()?;
449            let ident = s.value();
450            let Some(value) = convert_field_type_to_column_data_type(&ident) else {
451                return Err(meta.error(format!("unexpected {META_KEY_DATATYPE}: {ident}")));
452            };
453            attribute.column_data_type = Some(value);
454        }
455        META_KEY_SEMANTIC => {
456            let value = meta.value()?;
457            let s: LitStr = value.parse()?;
458
459            let ident = s.value();
460            let Some(value) = semantic_type_from_ident(&ident) else {
461                return Err(meta.error(format!("unexpected {META_KEY_SEMANTIC}: {ident}")));
462            };
463            attribute.semantic_type = value;
464        }
465        attr => return Err(meta.error(format!("unexpected attribute: {attr}"))),
466    }
467
468    Ok(())
